#include<stdio.h>
int main()
{
 int Size, i, Search, Flag;

 scanf("%d",&Size);
 int arr[Size];
 for(i = 0; i < Size; i++)
 {
 scanf("%d",&arr[i]);
 }
 scanf("%d",&Search);
Flag = 0;
 for(i = 0; i < Size; i++)
 {
 if(arr[i] == Search)
 {
 Flag = 1;
 break;
}
 }

 if(Flag == 1)
 {
 printf("We found the Search Element %d at Position %d ", Search, i + 1);
}
else
{
printf("Sorry!! We haven't found the the Search Element %d ", Search);
}
 return 0;
}
